requirejs(["module"], function(test) {
  window.addEventListener("load", function() {
    var word = test.capitalize(test.garble(prompt("coconut")))
    document.body.innerText = word
  })
})

define(function() {
  return {
    capitalize: function(word) {
      return word.charAt(0).toUpperCase() + word.slice(1)
    },

    garble: function(word) {
      return word.replace(/[auiyoe]/g, function() {
        return "auiyoe".charAt(Math.floor(Math.random() * 6))
      })
    }
  }
})

(function(root, mod) {
  if (typeof exports == "object" && typeof module == "object") // CommonJS
    return mod(exports, require("module"))
  else if (typeof define == "function" && define.amd) // AMD
    return define(["exports", "module"], mod)
  else // Plain browser env
    mod(root.multimodule = {}, root.module)
})(this, function(exports, module) {
